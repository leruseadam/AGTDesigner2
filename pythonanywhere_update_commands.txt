# Commands to run on PythonAnywhere bash console:

# 1. Navigate to your project directory
cd ~/AGTDesigner

# 2. Pull the latest changes from GitHub
git pull origin main

# 3. Clear Python cache
rm -rf __pycache__
rm -rf src/__pycache__
rm -rf src/core/__pycache__
rm -rf src/core/data/__pycache__
rm -rf src/core/generation/__pycache__

# 4. Set proper permissions
chmod -R 755 .
chmod -R 755 uploads output cache logs static

# 5. Go to PythonAnywhere Web tab and click "Reload" for your web app

# bash update_pythonanywhere.sh
